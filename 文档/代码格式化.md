### 基于eslint格式化
```js
//配置
"format": "eslint src/**/*.{jsx,js} --fix"
//这里会校验eslit自带的
//2、增加推荐的
 "eslintConfig": {
    "extends": [
      "eslint:recommended",
      "react-app"
    ],
    "rules": {
      "react/jsx-indent": ["error", 4],
      "no-console":"off"
    }
  },
```
### 提交代码自动格式化
```js
//1、npm i husky lint-staged -d
//husky :自动配置git-hooks
//lint-staged:给husky调用---针对本次修改的文件调用
//配置
"husky": {
    "hooks": {
      "pre-commit": "lint-staged"
    }
  },
  "lint-staged": {
    "*.{js,jsx}": [
      "prettier --write",
      "eslint --fix",//格式化
      "git add"//保存
    ],
    "*.{css,md}": [
      "prettier --write",
      "git add"
    ]
  },
```

### 基于prettier 格式化
```js
"format": "prettier src/**/*.{js,jsx,css,md} --write && eslint src/**/*.{jsx,js} --fix"

"lint-staged": {
    "*.{js,jsx}": [
      "prettier --write",

      "git add"
```
